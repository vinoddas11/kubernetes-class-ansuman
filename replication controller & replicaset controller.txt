                        Kubernetes - Replication Controller


apiVersion: v1
kind: ReplicationController
metadata:
  name: skillrary
spec:
  replicas: 3
  selector:
    app: nginx
  template:
    metadata:
      name: abc
      labels:
        app: nginx
    spec:
      containers:
      - name: xyz
        image: nginx
        ports:
        - containerPort: 80

                             Kubernetes - Replica Sets



apiVersion: apps/v1
kind: ReplicaSet
metadata:
  name: frontend
  labels:
    app: guestbook
    tier: frontend
spec:                      
  replicas: 3
  selector:
    matchLabels:
      tier: frontend
  template:
    metadata:
      labels:
        tier: frontend
    spec:
      containers:
      - name: php-redis
        image: gcr.io/google_samples/gb-frontend:v3




kind: ReplicaSet                                    
apiVersion: apps/v1                            
metadata:
  name: rs.pod
spec:
  replicas: 2  
  selector:                  
    matchExpressions:                             
      - {key: myname, operator: In, values: [qspider, jspider, pyspider]}
      - {key: env, operator: NotIn, values: [skillrary]}
  template:      
    metadata:
      name: anshuman
      labels:              
        myname: qspider
    spec:
     containers:
       - name: abcd
         image: ubuntu
         command: ["/bin/bash", "-c", "while true; do echo Hello-world; sleep 3 ; done"]
